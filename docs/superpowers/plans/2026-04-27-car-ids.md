# Car IDS Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a real-time intrusion detection system for an ESP32-based RC car that detects and corrects traffic signal spoofing, speed injection, and GPS spoofing attacks — with a live monitoring dashboard.

**Architecture:** ESP32 car firmware publishes sensor data over MQTT (WiFi) to a Mosquitto broker on a laptop. A Python IDS service subscribes to all topics, runs Isolation Forest for speed anomalies and Haversine-based checks for GPS integrity, then publishes corrections back to the car. A Flask+Socket.IO dashboard surfaces all events live. A hacker Python CLI injects attacks for demo purposes.

**Tech Stack:** Arduino/C++ (ESP32), Python 3.11+, paho-mqtt 1.6, scikit-learn 1.4, Flask 3.0, Flask-SocketIO 5.3, Chart.js 4.x, Mosquitto 2.x, NewPing, PubSubClient Arduino library

```cpp
void mqttCallback(char* topic, byte* payload, unsigned int length) {
  String msg = "";
```

```cpp
const unsigned long GREEN_DURATION = 5000;

void reconnect() {
  while (!mqtt.connected()) {
    if (mqtt.connect("traffic_ctrl")) {
      Serial.println("Traffic controller connected");
    } else {
      delay(2000);
    }
  }
}

void setup() {
  Serial.begin(115200);
  pinMode(IR_LED_PIN, OUTPUT);
  digitalWrite(IR_LED_PIN, LOW);

  WiFi.begin(WIFI_SSID, WIFI_PASS);
  while (WiFi.status() != WL_CONNECTED) { delay(500); Serial.print("."); }
  mqtt.setServer(MQTT_BROKER, MQTT_PORT);
  reconnect();
}

void loop() {
  if (!mqtt.connected()) reconnect();
  mqtt.loop();

  unsigned long duration = isRed ? RED_DURATION : GREEN_DURATION;
  if (millis() - lastToggle >= duration) {
    lastToggle = millis();
    isRed = !isRed;

    // Publish MQTT state

    mqtt.publish("traffic/signal", state);

    // Drive IR LED (only when RED — car stops on IR detection)
    digitalWrite(IR_LED_PIN, isRed ? HIGH : LOW);

    Serial.println(isRed ? "RED" : "GREEN");
  }
}
```

## Task 8: ML Training Script

**Files:**
- Create: `scripts/train_model.py`

- [ ] **Step 1: Write train_model.py**

```python
import pandas as pd
import numpy as np
from sklearn.ensemble import IsolationForest
from sklearn.preprocessing import StandardScaler
import joblib
import os

DATA_FILE  = "data/normal_drive.csv"
MODEL_DIR  = "models"
WINDOW     = 10   # 10 readings = 1 second at 100ms

os.makedirs(MODEL_DIR, exist_ok=True)

df = pd.read_csv(DATA_FILE)
pwm = df["pwm"].values

# Build feature matrix from sliding windows
def extract_features(window):
    w = np.array(window)
    deltas = np.diff(w)
    return [
        float(np.mean(w)),           # mean_pwm
        float(np.std(w)),            # std_pwm
        float(np.max(np.abs(deltas))) if len(deltas) else 0,  # max_delta_pwm
        float(np.mean(np.abs(deltas))) if len(deltas) else 0, # accel_rate
        float(np.max(w) - np.min(w)),                          # window_range
    ]

X = [extract_features(pwm[i:i+WINDOW]) for i in range(len(pwm) - WINDOW)]
X = np.array(X)

scaler = StandardScaler()
X_scaled = scaler.fit_transform(X)

model = IsolationForest(n_estimators=100, contamination=0.05, random_state=42)
model.fit(X_scaled)

# Determine threshold at 5th percentile of training scores
scores = model.decision_function(X_scaled)
threshold = float(np.percentile(scores, 5))
print(f"Trained on {len(X)} windows. Threshold: {threshold:.4f}")

joblib.dump(model,   f"{MODEL_DIR}/ids_model.pkl")
joblib.dump(scaler,  f"{MODEL_DIR}/scaler.pkl")
with open(f"{MODEL_DIR}/threshold.txt", "w") as f:
    f.write(str(threshold))

print(f"Model saved to {MODEL_DIR}/")
```

```python
from haversine import haversine, Unit

class GpsChecker:
    def __init__(self, golden_route: list[tuple], threshold_m: float = 50.0):
        self.route        = golden_route
        self.threshold_m  = threshold_m
        self.current_index = 0
        self.locked       = False

    def check(self, lat: float, lon: float) -> dict:
        if self.locked:
            return {"status": "locked", "deviation_m": None}

        expected    = self.route[self.current_index]
        deviation_m = haversine(expected, (lat, lon), unit=Unit.METERS)

        if deviation_m > self.threshold_m:
            self.locked = True
            return {"status": "intrusion", "deviation_m": round(deviation_m, 1)}

        # Advance to next waypoint on valid check
        self.current_index = (self.current_index + 1) % len(self.route)
        return {"status": "ok", "deviation_m": round(deviation_m, 1)}
```

```python
import numpy as np
from collections import deque
import joblib
import os

WINDOW_SIZE = 10

def _extract_features(window: list[float]) -> list[float]:
    w = np.array(window)
    deltas = np.abs(np.diff(w))
    return [
        float(np.mean(w)),
        float(np.std(w)),
        float(np.max(deltas)) if len(deltas) else 0.0,
        float(np.mean(deltas)) if len(deltas) else 0.0,
        float(np.max(w) - np.min(w)),
    ]

class SpeedIDS:
    def __init__(self, model_path: str | None, scaler_path: str | None,
                 threshold: float = -0.1):
        self.window          = deque(maxlen=WINDOW_SIZE)
        self.last_safe_pwm   = 80
        self.anomaly_counter = 0
        self.threshold       = threshold

        self.model  = joblib.load(model_path)  if model_path  and os.path.exists(model_path)  else None
        self.scaler = joblib.load(scaler_path) if scaler_path and os.path.exists(scaler_path) else None

    def _is_anomaly(self, features: list[float]) -> bool:
        # Fallback: z-score on max_delta_pwm (index 2)
        if self.model is None:
            return features[2] > 80  # delta > 80 PWM units in one step

        X = np.array([features])
        if self.scaler:
            X = self.scaler.transform(X)
        score = self.model.decision_function(X)[0]
        return score < self.threshold

    def update(self, pwm: float) -> dict:
        self.window.append(pwm)

        if len(self.window) < WINDOW_SIZE:
            return {"status": "collecting", "correction_pwm": None}

        features = _extract_features(list(self.window))

        if self._is_anomaly(features):
            self.anomaly_counter += 1
            if self.anomaly_counter >= 3:
                return {"status": "intrusion", "correction_pwm": self.last_safe_pwm}
            return {"status": "suspicious", "correction_pwm": None}
        else:
            self.anomaly_counter = 0
            if 20 < pwm < 240:
                self.last_safe_pwm = int(pwm)
            return {"status": "ok", "correction_pwm": None}
```

```python
class TrafficIDS:
    def check(self, ir_red: bool, mqtt_red: bool) -> dict:
        if ir_red == mqtt_red:

        # Mismatch — determine which channel was spoofed

        else:
```

```python
    global alert_cb
    alert_cb = cb

def _publish_alert(client, alert: dict):
    client.publish("ids/alert", json.dumps(alert))
    if alert_cb:
        alert_cb(alert)
    print(f"[IDS ALERT] {alert}")

def on_message(client, userdata, msg):
    topic   = msg.topic
    payload = msg.payload.decode()

    try:
        data = json.loads(payload)
    except json.JSONDecodeError:
        return

    if topic == "car/speed":
        pwm    = data.get("pwm", 0)
        result = speed_ids.update(pwm)
        if result["status"] == "intrusion":
            _publish_alert(client, {
                "type":    "speed_intrusion",
                "pwm":     pwm,
                "correct": result["correction_pwm"],
            })
            correction = json.dumps({
                "action": "set_speed",
                "value":  result["correction_pwm"],
            })
            client.publish("car/command", correction)

    elif topic == "car/gps":
        lat, lon = data.get("lat", 0), data.get("lon", 0)
        result   = gps_checker.check(lat, lon)
        if result["status"] in ("intrusion", "locked"):
            _publish_alert(client, {
                "type":        "gps_spoof",
                "lat":         lat,
                "lon":         lon,
                "deviation_m": result["deviation_m"],
            })

    elif topic == "car/ir":
        ir_state["ir_red"]   = data.get("ir_red",   False)
        ir_state["mqtt_red"] = data.get("mqtt_red", False)
        result = traffic_ids.check(
            ir_red=ir_state["ir_red"],
            mqtt_red=ir_state["mqtt_red"],
        )
        if result["status"] == "intrusion":
            _publish_alert(client, {
                "type":        "traffic_intrusion",
                "attack_type": result["attack_type"],
            })

    elif topic in ("hacker/speed", "hacker/gps", "hacker/ir"):
        # Log hacker activity — do not act on it
        print(f"[HACKER DETECTED] {topic}: {payload}")
        _publish_alert(client, {
            "type":    "hacker_activity",
            "topic":   topic,
            "payload": payload,
        })

def run():
    client = mqtt.Client()
    client.on_message = on_message
    client.connect(BROKER, PORT)
    client.subscribe([
        ("car/speed", 0), ("car/gps", 0), ("car/ir", 0),
        ("hacker/speed", 0), ("hacker/gps", 0), ("hacker/ir", 0),
    ])
    print("[IDS] Service running...")
    client.loop_forever()

if __name__ == "__main__":
    run()
```

### Bonus B: HMAC Signing on car/command

```python
# In ids_service.py — sign corrections before publishing
import hmac, hashlib, base64

SECRET = b"car-ids-secret-key"

def sign_payload(payload: str) -> str:
    sig = hmac.new(SECRET, payload.encode(), hashlib.sha256).digest()
    return payload + "." + base64.b64encode(sig).decode()
```
